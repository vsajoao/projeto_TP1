Este projeto tem como objetivo a implementação de um sistema para aprendizado sobre investimentos financeiros, utilizando a linguagem C++. O sistema permite a criação e gerenciamento de contas, carteiras e ordens de investimento, seguindo uma arquitetura em camadas.\hypertarget{md__r_e_a_d_m_e_autotoc_md1}{}\doxysection{\texorpdfstring{📌 Funcionalidades}{📌 Funcionalidades}}\label{md__r_e_a_d_m_e_autotoc_md1}

\begin{DoxyItemize}
\item {\bfseries{Gerenciamento de conta}}
\begin{DoxyItemize}
\item Criar, editar, excluir e visualizar conta.
\item Autenticação via CPF e senha.
\end{DoxyItemize}
\item {\bfseries{Gerenciamento de carteiras}}
\begin{DoxyItemize}
\item Criar, editar, excluir e listar carteiras associadas à conta.
\item Cada carteira tem um código, nome e perfil (Conservador, Moderado, Agressivo).
\end{DoxyItemize}
\item {\bfseries{Gerenciamento de ordens}}
\begin{DoxyItemize}
\item Criar e excluir ordens de investimento.
\item Cada ordem contém um código, código de negociação, data, preço e quantidade.
\item O preço da ordem é calculado com base no preço médio do papel na data informada.
\item Ordens não podem ser editadas, apenas excluídas.
\end{DoxyItemize}
\end{DoxyItemize}\hypertarget{md__r_e_a_d_m_e_autotoc_md2}{}\doxysection{\texorpdfstring{📂 Estrutura do Código}{📂 Estrutura do Código}}\label{md__r_e_a_d_m_e_autotoc_md2}

\item {\bfseries{Camada de serviço}} ~\newline

\end{DoxyItemize}
